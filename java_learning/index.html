package linkList_string;

public class linkListString {
    ListNode head;
    ListNode tail;
    int size;

    public linkListString() {
        head = null;
        tail = null;
        size = 0;
    }

    public void add(char val) {
        ListNode node = new ListNode(val);
        if (head == null) {
            head = node;
            tail = node;
        } else {
            tail.next = node;
            tail = node;
        }
        size++;
    }
}
